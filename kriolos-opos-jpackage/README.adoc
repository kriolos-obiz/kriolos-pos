== jdeps 


----
jdeps --print-module-deps --ignore-missing-deps --recursive --multi-release 21 --class-path="./target/lib/" --module-path="./target/lib/" ./target/lib/kriolos-opos-app-1.0.0-SNAPSHOT.jar
----


----
jdeps --list-deps --ignore-missing-deps --multi-release 21 --class-path './target/lib/*' ./target/lib/kriolos-opos-app-1.0.0-SNAPSHOT.jar
----
